ERD


[actor]       [film]         [category]
   |            |                |
   |            |                |
   |         [film_category]     |
   |__________/        \_________|
        many-to-many

[film] ──────< [inventory] >────── [store] (optional)

[inventory] ──────< [rentat] (rental) >────── [customer]
                                 |
                             [payment]

[customer] >── [address] >── [city] >── [country]

[film] >── [language]


DB Diagram

Table actor {
  actor_id int [pk]
  first_name varchar
  last_name varchar
  last_update datetime
}

Table film {
  film_id int [pk]
  title varchar
  description text
  release_year int
  language_id int [ref: > language.language_id]
  rental_duration int
  rental_rate decimal
  replacement_cost decimal
  rating varchar
  last_update datetime
}

Table language {
  language_id int [pk]
  name varchar
  last_update datetime
}

Table inventory {
  inventory_id int [pk]
  film_id int [ref: > film.film_id]
  store_id int
  last_update datetime
}

Table rentat {
  rental_id int [pk]
  rental_date datetime
  inventory_id int [ref: > inventory.inventory_id]
  customer_id int [ref: > customer.customer_id]
  return_date datetime
  staff_id int
  last_update datetime
}

Table payment {
  payment_id int [pk]
  customer_id int [ref: > customer.customer_id]
  rental_id int [ref: > rentat.rental_id]
  amount decimal
  payment_date datetime
}

Table customer {
  customer_id int [pk]
  store_id int
  first_name varchar
  last_name varchar
  email varchar
  address_id int [ref: > address.address_id]
  active bool
  create_date datetime
  last_update datetime
}

Table address {
  address_id int [pk]
  address varchar
  address2 varchar
  district varchar
  city_id int [ref: > city.city_id]
  postal_code varchar
  phone varchar
  last_update datetime
}

Table city {
  city_id int [pk]
  city varchar
  country_id int [ref: > country.country_id]
  last_update datetime
}

Table country {
  country_id int [pk]
  country varchar
  last_update datetime
}

Table category {
  category_id int [pk]
  name varchar
  last_update datetime
}
